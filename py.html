Reverse a number
num = int(input("Enter num"))
rev = 0
while num > 0:
    rem = num % 10
    rev = (rev * 10) + rem
    num //= 10
print(rev)


Palindrome of a number
num = int(input("Enter the number:___"))
rev = 0
temp = num
while temp != 0:
    rem = temp % 10
    rev = rev * 10 + rem
    temp = temp // 10
if num == rev:
    print("Palindrome")
else:
    print("Not Palindrome")


Reverse a given string
a = input("Enter a string:_________")
revstr = ""
i = len(a) - 1
while i >= 0:
    revstr = revstr + a[i]
    i -= 1
print(revstr)
